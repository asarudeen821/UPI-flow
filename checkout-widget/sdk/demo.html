💻 Integration Code

<!-- 1. Include SDK -->
<script src="https://cdn.yourdomain.com/checkout.js"></script>

<!-- 2. Add container -->
<div id="checkout-widget"></div>

<!-- 3. Initialize -->
<script>
  Checkout.init({
    publicKey: 'pk_test_xxx',
    amount: 499,
    currency: 'INR',
    productName: 'Premium Plan',
    gateway: 'razorpay',
    methods: ['upi', 'card', 'wallet'],
    customer: {
      name: 'John Doe',
      email: 'john@example.com',
      phone: '9876543210'
    },
    // Event handlers
    onPaymentSuccess: (data) => console.log('Success:', data),
    onPaymentFailed: (error) => console.error('Failed:', error)
  });
</script>
